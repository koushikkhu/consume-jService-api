-- Consume jService API --

URI: /clues/clue/{id}
Method: GET
Description: returns a clue having id={id}

URI: /clues/airdate/{airdate}
Method: GET
Description: returns all clues by airdate

URI: /clues/categoryid/{categoryid}
Method: GET
Description: returns all clues by categoryid

URI: /categories
Method: GET
Description: returns all clues

URI: /categories/category/{categoryid}
Method: GET
Description: returns a category having categoryid = {categoryid}
